Nominale Worst-Case-Rendite berechnen und in der Kachel anzeigen

Oberhalb des JSX-Returns (unterhalb deiner calculateScenarios()-Berechnung) füge folgende Hilfsfunktionen ein:

// Nominaler Gesamtbetrag der Einzahlungen (ohne Inflationseffekte auf die Beträge)
function computeNominalTotalPaid(inputs: CalculatorInputs): number {
  const years = inputs.targetAge - inputs.currentAge;
  const months = years * 12;
  let total = inputs.oneTimeAmount;

  for (let m = 1; m <= months; m++) {
    const yearIndex = Math.floor((m - 1) / 12);
    const monthly = inputs.monthlyAmount * Math.pow(1 + inputs.annualIncrease / 100, yearIndex);
    total += monthly;
  }
  return total;
}

// Approx. monatliche IRR über Bisection (nominal)
// Cashflows im Monatsabstand; negatives Vorzeichen = Einzahlung (Outflow)
function approxIrrMonthly(cashflows: number[], maxIter = 80, tol = 1e-8): number {
  let lo = -0.9999; // untere Schranke (nahe -100%/Monat)
  let hi = 1.0;     // obere Schranke (+100%/Monat)

  const npv = (r: number) => {
    let v = 0;
    for (let t = 0; t < cashflows.length; t++) {
      v += cashflows[t] / Math.pow(1 + r, t);
    }
    return v;
  };

  let npvLo = npv(lo);
  let npvHi = npv(hi);
  // Falls gleiches Vorzeichen, versuche den Intervall etwas zu erweitern
  if (npvLo * npvHi > 0) {
    // Fallback: CAGR-Annäherung über totalPaid und Endwert (nur sinnvoll bei Einmalanlage)
    return NaN;
  }

  for (let i = 0; i < maxIter; i++) {
    const mid = 0.5 * (lo + hi);
    const v = npv(mid);
    if (Math.abs(v) < tol) return mid;
    if (npvLo * v < 0) {
      hi = mid;
      npvHi = v;
    } else {
      lo = mid;
      npvLo = v;
    }
  }
  return 0.5 * (lo + hi);
}

// Baut nominale Monats-Cashflows: Einzahlungen negativ, am Ende nominales Worst-Case-Endvermögen positiv
function buildNominalCashflowsForWorstCase(inputs: CalculatorInputs, worstCaseEndNominal: number): number[] {
  const years = inputs.targetAge - inputs.currentAge;
  const months = years * 12;

  const cfs: number[] = [];
  // t=0: Einmalbetrag (Outflow)
  if (inputs.oneTimeAmount && inputs.oneTimeAmount > 0) cfs.push(-inputs.oneTimeAmount);
  else cfs.push(0);

  // t=1..months-1: monatliche Einzahlungen mit Jahressteigerung (nominal)
  for (let m = 1; m < months; m++) {
    const yearIndex = Math.floor((m - 1) / 12);
    const monthly = inputs.monthlyAmount * Math.pow(1 + inputs.annualIncrease / 100, yearIndex);
    cfs.push(-monthly);
  }

  // t=months: letzte Einzahlung + Endwert (positiv)
  if (months > 0) {
    const yearIndex = Math.floor((months - 1) / 12);
    const monthlyLast = inputs.monthlyAmount * Math.pow(1 + inputs.annualIncrease / 100, yearIndex);
    cfs.push(-monthlyLast + worstCaseEndNominal);
  }

  return cfs;
}


Berechne daraus die nominale p.a.-Rendite für den Worst-Case-Wert:

// Nominale Worst-Case p.a. Rendite
const worstCaseEndNominal = result.worstCase; // Endwert Worst Case (nominal)
const yearsTotal = inputs.targetAge - inputs.currentAge;

// Cashflows nominal auf Monatsbasis (mit Monatsraten)
const cfs = buildNominalCashflowsForWorstCase(inputs, worstCaseEndNominal);
let worstCaseIrrPaNominal: number | null = null;

if (inputs.monthlyAmount > 0) {
  const irrMonthly = approxIrrMonthly(cfs);
  if (!Number.isNaN(irrMonthly)) {
    worstCaseIrrPaNominal = Math.pow(1 + irrMonthly, 12) - 1;
  }
} else {
  // Fallback: CAGR für reine Einmalanlage
  const totalPaidNominal = computeNominalTotalPaid(inputs); // = oneTimeAmount in diesem Pfad
  if (totalPaidNominal > 0) {
    worstCaseIrrPaNominal = Math.pow(worstCaseEndNominal / totalPaidNominal, 1 / yearsTotal) - 1;
  }
}

// Formatierer
const formatPct = (x: number | null) => x === null ? '–' : `${(x * 100).toFixed(2)}% p.a.`;
